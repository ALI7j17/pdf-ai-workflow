4. مثال كود أساسي لتحليل وتصحيح نص من PDF

python
import PyPDF2
import language_tool_python

def extract_text_from_pdf(pdf_path):
    with open(pdf_path, 'rb') as file:
        reader = PyPDF2.PdfReader(file)
        text = ''
        for page in reader.pages:
            text += page.extract_text()
        return text

def correct_text(text):
    tool = language_tool_python.LanguageTool('en-US')
    matches = tool.check(text)
    corrected_text = language_tool_python.utils.correct(text, matches)
    return corrected_text

if __name__ == '__main__':
    pdf_text = extract_text_from_pdf("sample.pdf")
    corrected = correct_text(pdf_text)
    print(corrected)

✨ تحديث كود analyze_pdf.py ليتوافق مع هذا التنظيم:

python
import os
import PyPDF2
import language_tool_python

# المسارات
input_folder = "input"
output_folder = "output"
input_file = "sample.pdf"  # اسم الملف فقط
output_file = "sample_corrected.txt"

def extract_text_from_pdf(file_path):
    text = ""
    with open(file_path, "rb") as file:
        reader = PyPDF2.PdfReader(file)
        for page in reader.pages:
            text += page.extract_text() or ""
    return text

def correct_text(text):
    tool = language_tool_python.LanguageTool('en-US')
    matches = tool.check(text)
    corrected = language_tool_python.utils.correct(text, matches)
    return corrected

if __name__ == "__main__":
    # المسار الكامل للملف
    input_path = os.path.join(input_folder, input_file)
    output_path = os.path.join(output_folder, output_file)

    # استخراج وتصحيح
    extracted = extract_text_from_pdf(input_path)
    corrected = correct_text(extracted)

    # حفظ الناتج
    with open(output_path, "w", encoding="utf-8") as f:
        f.write(corrected)

    print(f"✅ النص المصحّح تم حفظه في: {output_path}")
